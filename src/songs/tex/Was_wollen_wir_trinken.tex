\verse{
\li{\brep\Em[]Was wollen wir trinken, sieben Tage \D[]lang?}
\li{Was wollen wir \C[]trinken? \D[]So ein \Em[]Durst!\erep}
\li{\brep Es wird ge\G[]nug für \D[]alle \G[]sein,}
\li{wir trinken zu\Em[]sammen, rollt das Fass mal \D[]rein!}
\li{Wir trinken zu\C[]sammen, \D[]nicht al\Em[]lein.\erep}
}

\verse{
\li{\brep\Em[]Dann wollen wir schaffen, sieben Tage \D[]lang!}
\li{Dann wollen wir \C[]schaffen, \D[]komm fass \Em[]an.\erep}
\li{\brep Und das wird \G[]keine \D[]Placke\G[]rei!}
\li{Wir schaffen zu\Em[]sammen, sieben Tage \D[]lang!}
\li{Ja, schaffen zu\C[]sammen, \D[]nicht al\Em[]lein\erep.}
}

\verse{
\li{\brep\Em[]Erst müssen wir kämpfen! Keiner weiß wie \D[]lang.}
\li{Erst müssen wir \C[]kämpfen, für \D[]unser \Em[]Ziel.\erep}
\li{\brep Und für das \G[]Glück von \D[]Jeder\G[]mann,}
\li{dafür heißt es \Em[]kämpfen, los fangt heute \D[]an!}
\li{Wir kämpfen zu\C[]sammen, \D[]nicht al\Em[]lein.\erep}
}

\verse{
\li{\brep\Em[]Jetzt müssen wir streiten, keiner weiß wie \D[]lang.}
\li{Ja, für ein \C[]Leben \D[]ohne \Em[]Zwang!\erep}
\li{\brep Dann kriegt der \G[]Frust uns \D[]nicht mehr \G[]klein!}
\li{Wir halten zu\Em[]sammen, keiner kämpft al\D[]lein!}
\li{Wir gehen zu\C[]sammen, \D[]nicht al\Em[]lein!\erep}
}

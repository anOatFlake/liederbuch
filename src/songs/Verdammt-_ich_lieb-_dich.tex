\verse{
\li{\Am Ich ziehe durch die Straßen bis nach Mitternacht,}
\li{'hab das früher auch gern gemacht, dich \G brauch ich - dafür \Am nicht.}
\li{\Am Ich sitz' am Tresen, trinke noch 'n Bier,}
\li{früher war'n wir oft gemeinsam hier, das \G macht mir, macht mir \Am nichts.}
\li{Gegen\C[]über sitzt 'n Typ wie 'n Bär, ich \G stell' mir vor, wenn das dein Neuer wär',}
\li{das \Am juckt mich - \G überhaupt \Am nicht.}
\li{Auf \C einmal packt's mich, ich geh auf ihn zu und \G mach ihn an: \dq Lass meine Frau in Ruh'!\dq}
\li{Er \Am fragt nur: \dq\G Hast du 'n \Am Stich?\dq}
\li{Und ich \F denke schon wieder nur an \G dich.}
